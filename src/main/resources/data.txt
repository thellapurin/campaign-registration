

CREATE TABLE campaign_registration
(
    campaignid integer NOT NULL DEFAULT nextval('campaign_registration_campaignid_seq'::regclass),
    message jsonb,
    campaign_approval boolean,
    campaign_classfification integer,
    campaign_name character varying(255) COLLATE pg_catalog."default",
    campaignkind integer,
    compensation integer,
    delivery_package_kind integer,
    description character varying(255) COLLATE pg_catalog."default",
    end_date date,
    evaluation_campaign_name character varying(255) COLLATE pg_catalog."default",
    evaluation_campaignid integer,
    execution_date character varying(255) COLLATE pg_catalog."default",
    filtering_conditions_type integer,
    mechanical_required boolean,
    nickname character varying(255) COLLATE pg_catalog."default",
    notification_number character varying(255) COLLATE pg_catalog."default",
    rxswin character varying(255) COLLATE pg_catalog."default",
    rxswin_update_info_name character varying(255) COLLATE pg_catalog."default",
    rxswin_updateinfold integer,
    rxswin_updateinfos jsonb,
    update_approval boolean,
    update_usernotificationinfo_name character varying(255) COLLATE pg_catalog."default",
    updated_usernotificationinfold integer,
    CONSTRAINT campaign_registration_pkey PRIMARY KEY (campaignid)
);
